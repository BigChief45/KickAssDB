CREATE TABLE Person (
person_id INT PRIMARY KEY,
person_name VARCHAR(20),
person_last VARCHAR(30),
age INT,
salary INT,
birthday VARCHAR(10),
comment VARCHAR(30),
nationality VARCHAR(20),
security int
);

CREATE TABLE Vehicle (
vehicle_id INT PRIMARY KEY,
plate_number VARCHAR(20),
model VARCHAR(30)
);
